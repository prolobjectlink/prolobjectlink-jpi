33  package io.github.prolobjectlink.prolog;

35  import java.util.List;
36  import java.util.Map;

38  /**
39   * Converter for convert {@code PrologTerm} to the equivalent Java object taking
40   * like reference the following equivalence table.
41   * 
42   * <table BORDER > <caption>Equivalence table</caption>
43   * <tr>
44   * <td ALIGN=CENTER><b>Java Object</b></td>
45   * <td ALIGN=CENTER><b>Prolog Term</b></td>
46   * </tr>
47   * <tr>
48   * <td><code>null</code></td>
49   * <td>{@link PrologProvider#prologNil()}</td>
50   * </tr>
51   * <tr>
52   * <td>{@link String}</td>
53   * <td>{@link PrologAtom}</td>
54   * </tr>
55   * <tr>
56   * <td>{@link Boolean#FALSE}</td>
57   * <td>{@link PrologProvider#prologFalse()}</td>
58   * </tr>
59   * <tr>
60   * <td>{@link Boolean#TRUE}</td>
61   * <td>{@link PrologProvider#prologTrue()}</td>
62   * </tr>
63   * <tr>
64   * <td>{@link Integer}</td>
65   * <td>{@link PrologInteger}</td>
66   * </tr>
67   * <tr>
68   * <td>{@link Float}</td>
69   * <td>{@link PrologFloat}</td>
70   * </tr>
71   * <tr>
72   * <td>{@link Double}</td>
73   * <td>{@link PrologDouble}</td>
74   * </tr>
75   * <tr>
76   * <td>{@link Long}</td>
77   * <td>{@link PrologLong}</td>
78   * </tr>
79   * <tr>
80   * <td>{@link Object}[]</td>
81   * <td>{@link PrologList}</td>
82   * </tr>
83   * </table>
84   * 
85   * There are special cases of Java object that can be converted to Prolog
86   * equivalent but the inverse case it's not possible. They are {@link Byte},
87   * {@link Character}, {@link Short} that can be converted to
88   * {@link PrologInteger} using your numeric value. The main problems is that
89   * after {@link PrologInteger} conversion this value will be converted in
90   * {@link Integer}.
91   * 
92   * @author Jose Zalacain
93   * @since 1.0
94   */
95  public interface PrologJavaConverter {
96  
97  	/**
98  	 * Create an equivalent Java object map list using the given Prolog terms map
99  	 * array. The resulting map list contains maps that have the same string key and
100 	 * the value for every key is a conversion from Prolog term to Java object.
101 	 * 
102 	 * @param maps Prolog terms map list to be converted in Java objects map list.
103 	 * @return an equivalent Java objects map array using the given Prolog terms map
104 	 *         list.
105 	 * @since 1.0
106 	 */
107 	public List<Map<String, Object>> toObjectMaps(Map<String, PrologTerm>[] maps);
108 
109 	/**
110 	 * Create an equivalent Java object map using the given Prolog terms map. The
111 	 * resulting map have the same string key and the value for every key is a
112 	 * conversion from Prolog term to Java object.
113 	 * 
114 	 * @param map Prolog terms map representation to be converted
115 	 * @return an equivalent Java object map using the given Prolog terms map.
116 	 * @since 1.0
117 	 */
118 	public Map<String, Object> toObjectMap(Map<String, PrologTerm> map);
119 
120 	/**
121 	 * Create a Java objects list from given Prolog term array.
122 	 * 
123 	 * @param terms Prolog term array to be converted to Java objects array.
124 	 * @return Java objects list representation from given Prolog term array.
125 	 * @since 1.0
126 	 */
127 	public List<Object> toObjectList(PrologTerm[] terms);
128 
129 	/**
130 	 * Create an equivalent list of objects lists using the given Prolog terms
131 	 * matrix.
132 	 * 
133 	 * @param terms Prolog terms matrix to be converted
134 	 * @return an equivalent list of objects lists using the given Prolog terms
135 	 *         matrix representation.
136 	 * @since 1.0
137 	 */
138 	public List<List<Object>> toObjectLists(PrologTerm[][] terms);
139 
140 	/**
141 	 * Create a Java objects array from given Prolog term array.
142 	 * 
143 	 * @param terms Prolog term array to be converted to Java objects array.
144 	 * @return Java objects array representation from given Prolog term array.
145 	 * @since 1.0
146 	 */
147 	public Object[] toObjectsArray(PrologTerm[] terms);
148 
149 	/**
150 	 * Create an equivalent Prolog terms array using the given Java objects array.
151 	 * 
152 	 * @param objects Java objects array representation to be converted
153 	 * @return an equivalent Prolog terms array using the given Java objects array.
154 	 * @since 1.0
155 	 */
156 	public PrologTerm[] toTermsArray(Object[] objects);
157 
158 	/**
159 	 * Create a Java object from given Prolog term.
160 	 * 
161 	 * @param term Prolog term to be converted to Java object.
162 	 * @return a Java object from given Prolog term.
163 	 * @since 1.0
164 	 */
165 	public Object toObject(PrologTerm term);
166 
167 	/**
168 	 * Create an equivalent Prolog term using the given Java object.
169 	 * 
170 	 * @param object Java object to be converted
171 	 * @return an equivalent Prolog term using the given Java object.
172 	 * @since 1.0
173 	 */
174 	public PrologTerm toTerm(Object object);
175 
176 	/**
177 	 * Check if the current functor have quotes at the start and end of the given
178 	 * functor. Return true if the current functor is enclosing between functors or
179 	 * false in other case.
180 	 * 
181 	 * @param functor string functor to be checked
182 	 * @return true if the current functor is enclosing between functors or false in
183 	 *         other case.
184 	 * @since 1.0
185 	 */
186 	public boolean containQuotes(String functor);
187 
188 	/**
189 	 * Remove functor quotes if they are present. If the string functor don't have
190 	 * quotes, this method return the functor itself.
191 	 * 
192 	 * @param functor string functor to remove quotes if they are present.
193 	 * @return functor without quotes if they are present. The functor itself if
194 	 *         quotes are not present.
195 	 * @since 1.0
196 	 */
197 	public String removeQuotes(String functor);
198 
199 }
