35  import java.util.logging.Level;

38  /**
39   * Logger platform interface to log message at any level. Is an adapter for
40   * {@link Logger} adapting the Java logger mechanism for use with the most
41   * popular logger methods. This logger mechanism is accessible from
42   * {@link PrologProvider#getLogger()} or {@link PrologEngine#getLogger()} This
43   * logger interface have all traditional methods used to log messages at
44   * different levels (trace, debug, info,warn,error).
45   * 
46   * The levels used for this logger interface are {@link Level} constants present
47   * in the table.
48   * 
49   * <table BORDER > <caption>Level table</caption>
50   * <tr>
51   * <td ALIGN=CENTER><b>Method</b></td>
52   * <td ALIGN=CENTER><b>Level</b></td>
53   * </tr>
54   * <tr>
55   * <td>{@link #trace(Object, Object, Throwable)}</td>
56   * <td>{@link Level#FINEST}</td>
57   * </tr>
58   * <tr>
59   * <td>{@link #debug(Object, Object, Throwable)}</td>
60   * <td>{@link Level#FINE}</td>
61   * </tr>
62   * <tr>
63   * <td>{@link #info(Object, Object, Throwable)}</td>
64   * <td>{@link Level#INFO}</td>
65   * </tr>
66   * <tr>
67   * <td>{@link #warn(Object, Object, Throwable)}</td>
68   * <td>{@link Level#WARNING}</td>
69   * </tr>
70   * <tr>
71   * <td>{@link #error(Object, Object, Throwable)}</td>
72   * <td>{@link Level#SEVERE}</td>
73   * </tr>
74   * </table>
75   * 
76   * By default the platform implement a logger mechanism for drop log messages in
77   * Operating System temporal directory into files named
78   * prolobjectlink-YYYY.MM.DD.
79   * 
80   * In {@link AbstractLogger} class there are many implementations for this
81   * interface. Every final implementation class can extends from
82   * {@link AbstractLogger}.
83   * 
84   * @author Jose Zalacain
85   * @since 1.0
86   */
