35  /**
36   * Runtime error raised when occurs one call to get get indicator method over a
37   * term that no have indicator property. The indicator is a compound term and
38   * atom property only. This runtime error is raised when we invoke
39   * {@link PrologTerm#getIndicator()} over terms like Numbers and Variables
40   * 
41   * @author Jose Zalacain
42   * @since 1.0
43   */
44  public final class IndicatorError extends PrologError {
45  
46  	private static final long serialVersionUID = 3634744000779459116L;
47  
48  	/**
49  	 * Create an indicator runtime error. This constructor give the term over
50  	 * runtime error occurs.
51  	 * 
52  	 * @param term the term over runtime error occurs.
53  	 * @since 1.0
54  	 */
55  	public IndicatorError(Object term) {
56  		super("The term not have indicator: " + term);
57  	}
58  
59  }
