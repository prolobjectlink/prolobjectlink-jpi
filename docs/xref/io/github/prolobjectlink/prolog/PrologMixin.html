285 		final int prime = 31;
286 		int result = 1;
287 		result = prime * result + ancestors.hashCode();
288 		result = prime * result + methods.hashCode();
289 		result = prime * result + ((name == null) ? 0 : name.hashCode());

331 	@Override
332 	public String toString() {
333 		StringBuilder builder = new StringBuilder();
334 		builder.append(":-namespace(" + namesapce + ").");
335 		builder.append('\n');
336 		for (PrologTerm prologTerm : directives) {
337 			builder.append(":-" + prologTerm);
338 			builder.append('\n');
339 		}
340 		builder.append('\n');
341 		for (PrologMixin ancestor : ancestors) {
342 			builder.append(":-" + ancestor);
343 			builder.append('\n');
344 		}
345 		builder.append('\n');
346 		builder.append(name);
347 		builder.append(getLeftencloser());
348 		builder.append('\n');
349 		for (PrologClause method : methods) {
350 			builder.append(method);
351 			builder.append('\n');
352 		}
353 		builder.append('\n');
354 		builder.append(getRightencloser());
355 		return builder.toString();
356 	}
