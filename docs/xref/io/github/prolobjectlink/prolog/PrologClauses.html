37  /**

39   * indicator. Support all {@link List} operations to manage clauses in the
40   * family list.
41   * 
42   * @author Jose Zalacain
43   * @since 1.0
44   */
45  public interface PrologClauses extends List<PrologClause> {
46  
47  	/**
48  	 * True if this clause family list is a dynamic, false in other case
49  	 * 
50  	 * @deprecated Natives engine don't offer information about that.
51  	 * @return whether this clause family list is a dynamic
52  	 * @since 1.0
53  	 */
54  	@Deprecated
55  	public boolean isDynamic();
56  
57  	/**
58  	 * True if this clause family list is a multifile, false in other case
59  	 * 
60  	 * @deprecated Natives engine don't offer information about that.
61  	 * @return whether this clause family list is a multifile
62  	 * @since 1.0
63  	 */
64  	@Deprecated
65  	public boolean isMultifile();
66  
67  	/**
68  	 * True if this clause family list is a discontiguos, false in other case
69  	 * 
70  	 * @deprecated Natives engine don't offer information about that.
71  	 * @return whether this clause family list is a discontiguos.
72  	 * @since 1.0
73  	 */
74  	@Deprecated
75  	public boolean isDiscontiguous();
76  
77  	/**
78  	 * Clause family functor/arity based indicator. The clause family indicator is
79  	 * the same indicator for all clauses head in the clause family.
80  	 * 
81  	 * @return functor/arity based indicator of the current clause family.
82  	 * @since 1.0
83  	 */
84  	public String getIndicator();
85  
86  }
