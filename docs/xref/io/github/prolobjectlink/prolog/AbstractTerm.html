179 	public Object getObject() {
180 		PrologTerm term = this;
181 		switch (term.getType()) {
182 		case NIL_TYPE:
183 			return null;
184 		case CUT_TYPE:
185 			return "!";
186 		case FAIL_TYPE:
187 			return "fail";
188 		case TRUE_TYPE:
189 			return true;
190 		case FALSE_TYPE:
191 			return false;
192 		case ATOM_TYPE:
193 			return term.getFunctor();
194 		case FLOAT_TYPE:
195 			return ((PrologNumber) term).getFloatValue();
196 		case INTEGER_TYPE:
197 			return ((PrologNumber) term).getIntegerValue();
198 		case DOUBLE_TYPE:
199 			return ((PrologNumber) term).getDoubleValue();
200 		case LONG_TYPE:
201 			return ((PrologNumber) term).getLongValue();
202 		case VARIABLE_TYPE:
203 			return ((PrologVariable) term).getName();
204 		case LIST_TYPE:
205 			return fromTermArray(term.getArguments(), Object[].class);
206 		case STRUCTURE_TYPE:
207 			// NOTE if structure no have mapping
208 			// the term itself will be returned ???
209 			// Object object = term
210 			Object object = null;
211 			Collection<Prologable<?>> c = provider.values();
212 			for (Prologable<?> prologMapping : c) {
213 				PrologTerm mgu = prologMapping.toTerm(provider);
214 				if (mgu.unify(term)) {
215 					object = prologMapping.fromTerm(provider, term);
216 				}
217 			}
218 			return object;
219 		case OBJECT_TYPE:
220 			return term.getObject();
221 		case FIELD_TYPE:
222 			PrologField field = term.cast();
223 			return "field " + field.getName();
224 		case RESULT_TYPE:
225 			PrologResult result = term.cast();
226 			return "result " + result.getName();
227 		case PARAMETER_TYPE:
228 			PrologParameter parameter = term.cast();
229 			return "parameter " + parameter.getName();
230 		case CLASS_TYPE:
231 		case MIXIN_TYPE:
232 			// NOTE if structure no have mapping
233 			// the term itself will be returned ???
234 			// Object object = term
235 			return "class " + term.getFunctor();
236 		default:
237 			return null;
238 		}
239 	}

286 	 * Match to other term returning list of substitutions.
287 	 * 
288 	 * @param term - term to match check
289 	 * @return list of substitutions.
290 	 */
291 	public final Map<String, PrologTerm> match(PrologTerm term) {
292 		Deque<PrologTerm> stack = new ArrayDeque<PrologTerm>();
293 		if (unify(term, stack)) {
294 			int size = stack.size();
295 			HashMap<String, PrologTerm> substitution = new HashMap<String, PrologTerm>(size);
296 			while (size > 0) {
297 				PrologVariable variable = (PrologVariable) stack.pop();
298 				substitution.put(variable.getName(), variable.getTerm());
299 				// variable.unbind();
300 				size--;
301 			}
302 			return substitution;
303 		}
304 		return new HashMap<String, PrologTerm>();
305 	}

307 	/**
308 	 * Unification is the basic primitive operation in logic programming. Check that
309 	 * two terms (x and y) unify. Prolog unification algorithm is based on three
310 	 * principals rules:
311 	 * <ul>
312 	 * <li>If x and y are atomics constants then x and y unify only if they are same
313 	 * object.</li>
314 	 * <li>If x is a variable and y is anything then they unify and x is
315 	 * instantiated to y. Conversely, if y is a variable then this is instantiated
316 	 * to x.</li>
317 	 * <li>If x and y are structured terms then unify only if they match (equals
318 	 * funtor and arity) and all their correspondents arguments unify.</li>
319 	 * </ul>
320 	 * 
321 	 * 
322 	 * @param term  - the term to unify with the current term
323 	 * @param stack - the stack is used to store the addresses of variables which
324 	 *              are bound by the unification. This is needed when backtracking.
325 	 * @return true if the specified term unify whit the current term, false if not
326 	 */
327 	private boolean unify(PrologTerm term, Deque<PrologTerm> stack) throws PrologError {
328 
329 		PrologTerm thisTerm = this;
330 		PrologTerm otherTerm = term;
331 
332 		if (thisTerm.isVariableBound()) {
333 			return ((AbstractTerm) thisTerm.getTerm()).unify(otherTerm, stack);
334 		}
335 
336 		else if (otherTerm.isVariableBound()) {
337 			return ((AbstractTerm) otherTerm.getTerm()).unify(thisTerm, stack);
338 		}
339 
340 		// current term is a free variable
341 		else if (thisTerm.isVariableNotBound()) {
342 			// if (!thisTerm.occurs(otherTerm)) {
343 			// thisTerm.bind(otherTerm);
344 			stack.push(thisTerm);
345 			return true;
346 			// }
347 		}
348 
349 		// the other term is a free variable
350 		else if (otherTerm.isVariableNotBound()) {
351 			// if (!otherTerm.occurs(thisTerm)) {
352 			// otherTerm.bind(thisTerm);
353 			stack.push(otherTerm);
354 			return true;
355 			// }
356 		}
357 
358 		else {
359 
360 			int thisArity = thisTerm.getArity();
361 			int otherArity = otherTerm.getArity();
362 			String thisFunctor = thisTerm.getFunctor();
363 			String otherFunctor = otherTerm.getFunctor();
364 
365 			if (thisFunctor.equals(otherFunctor) && thisArity == otherArity) {
366 				for (int i = 0; i < thisArity; i++) {
367 					if (thisTerm.getArgument(i) != null && otherTerm.getArgument(i) != null) {
368 						if (!((AbstractTerm) thisTerm.getArgument(i)).unify(otherTerm.getArgument(i), stack)) {
369 							return false;
370 						}
371 					}
372 				}
373 				return true;
374 			}
375 		}
376 		return false;
377 	}
