53  public abstract class AbstractTerm implements PrologTerm {
54  
55  	protected int type;
56  	protected final PrologProvider provider;
57  
58  	protected AbstractTerm(int type, PrologProvider provider) {
59  		this.type = type;
60  		this.provider = provider;
61  	}
62  
63  	protected final void checkIndex(int index) {
64  		if (index < 0) {
65  			throw new ArrayIndexOutOfBoundsException(index);
66  		}
67  	}
68  
69  	protected final void checkIndex(int index, int max) {
70  		if (index < 0 || index > max) {
71  			throw new ArrayIndexOutOfBoundsException(index);
72  		}
73  	}
74  
75  	protected final String removeQuoted(String functor) {
76  		if (functor != null && functor.startsWith("\'") && functor.endsWith("\'")) {
77  			return functor.substring(1, functor.length() - 1);
78  		}
79  		return functor;
80  	}
81  
82  	protected final <K extends PrologTerm> K toTerm(Object o, Class<K> from) {
83  		return provider.toTerm(o, from);
84  	}
85  
86  	protected final <K extends PrologTerm> K[] toTermArray(Object[] os, Class<K[]> from) {
87  		return provider.toTermArray(os, from);
88  	}
89  
90  	protected final <K> K fromTerm(PrologTerm term, Class<K> to) {
91  		return provider.fromTerm(term, to);
92  	}
93  
94  	protected final <K> K[] fromTermArray(PrologTerm[] terms, Class<K[]> to) {
95  		return provider.fromTermArray(terms, to);
96  	}
97  
98  	protected final PrologLogger getLogger() {
99  		return provider.getLogger();
100 	}

102 	public PrologTerm getArgument(int index) {
103 		PrologTerm[] array = getArguments();

163 	public Object getObject() {
164 		PrologTerm term = this;
165 		switch (term.getType()) {
166 		case NIL_TYPE:
167 			return null;
168 		case CUT_TYPE:
169 			return "!";
170 		case FAIL_TYPE:
171 			return "fail";
172 		case TRUE_TYPE:
173 			return true;
174 		case FALSE_TYPE:
175 			return false;
176 		case ATOM_TYPE:
177 			return term.getFunctor();
178 		case FLOAT_TYPE:
179 			return ((PrologNumber) term).getFloatValue();
180 		case INTEGER_TYPE:
181 			return ((PrologNumber) term).getIntegerValue();
182 		case DOUBLE_TYPE:
183 			return ((PrologNumber) term).getDoubleValue();
184 		case LONG_TYPE:
185 			return ((PrologNumber) term).getLongValue();
186 		case VARIABLE_TYPE:
187 			return ((PrologVariable) term).getName();
188 		case LIST_TYPE:
189 			return fromTermArray(term.getArguments(), Object[].class);
190 		case STRUCTURE_TYPE:
191 			// NOTE if structure no have mapping
192 			// the term itself will be returned ???
193 			// Object object = term
194 			Object object = null;
195 			Collection<Prologable<?>> c = provider.values();
196 			for (Prologable<?> prologMapping : c) {
197 				PrologTerm mgu = prologMapping.toTerm(provider);
198 				if (mgu.unify(term)) {
199 					object = prologMapping.fromTerm(provider, term);
200 				}
201 			}
202 			return object;
203 		case OBJECT_TYPE:
204 			return term.getObject();
205 		default:
206 			return null;
207 		}
208 	}
