38  	 * Check if the given java class member (filed, method or constructor) is
39  	 * accessible or not.
40  	 * 
41  	 * @param member java member (filed, method or constructor) to be checked
42  	 * @return true if the given java class member is accessible, false if not.
43  	 * @since 1.0
44  	 */
45  	public static boolean isAccessible(AccessibleObject member) {
46  		return member.isAccessible();
47  	}
48  
49  	public static void doAccessibleIfNeed(AccessibleObject member) {
50  		if (!isAccessible(member)) {
51  			member.setAccessible(true);
52  		}
53  	}
54  
55  	public static void undoAccessibleIfNeed(AccessibleObject member) {
56  		if (!isAccessible(member)) {
57  			member.setAccessible(false);
58  		}
59  	}

71  	/**
72  	 * Find some constructor for create an empty instance. This method create an
73  	 * object instance under critical situations. There are non-empty declared
74  	 * constructor, restricted access for alternative constructor with any number
75  	 * and type of parameters. The alternative constructor with any number and type
76  	 * of parameter will be create with an array with the default value (0 for
77  	 * primitive types, false for boolean and null for any class type). This is the
78  	 * way for invoke {@link Constructor#newInstance(Object...)}. The public empty
79  	 * constructor simplify the situation because no have additional parameters
80  	 * analysis.
81  	 * 
82  	 * @param clazz type to be create an instance
83  	 * @return instance from given class.
84  	 * @since 1.0
85  	 */
86  	public static Object newInstance(Class<?> clazz) {
87  		Object instance = null;
88  		Constructor<?>[] constructors = clazz.getDeclaredConstructors();
89  		if (constructors.length > 0) {
90  			Constructor<?> constructor = constructors[0];
91  			Class<?>[] classes = constructor.getParameterTypes();
92  			Object[] arguments = new Object[classes.length];
93  			for (int i = 0; i < classes.length; i++) {
94  				if (classes[i] == String.class) {
95  					arguments[i] = "";
96  				} else if (classes[i] == boolean.class) {
97  					arguments[i] = false;
98  				} else if (classes[i].isPrimitive()) {
99  					arguments[i] = 0;
100 				}
101 			}
102 			try {
103 
104 				doAccessibleIfNeed(constructor);
105 				instance = constructor.newInstance(arguments);
106 				undoAccessibleIfNeed(constructor);
107 
108 			} catch (InstantiationException e) {
109 				Prolog.getProvider().getLogger().error(ReflectionUtil.class, e.getMessage(), e);
110 			} catch (IllegalArgumentException e) {
111 				Prolog.getProvider().getLogger().error(ReflectionUtil.class, e.getMessage(), e);
112 			} catch (IllegalAccessException e) {
113 				Prolog.getProvider().getLogger().error(ReflectionUtil.class, e.getMessage(), e);
114 			} catch (InvocationTargetException e) {
115 
116 				// This exception is reported when a constructor not
117 				// initialize directly yours fields, but make the job
118 
119 			}
120 
121 		}
122 		return instance;
123 	}

125 	public static <T> T newInstance(Class<T> cls, Class<?>[] parameterTypes, Object[] parameterValues) {
126 		T value = null;
127 		try {
128 			Constructor<T> constructor = cls.getDeclaredConstructor(parameterTypes);
129 			doAccessibleIfNeed(constructor);
130 			value = constructor.newInstance(parameterValues);
131 			undoAccessibleIfNeed(constructor);
132 		} catch (InstantiationException e) {
133 			Prolog.getProvider().getLogger().error(ReflectionUtil.class, e.getMessage(), e);
134 		} catch (IllegalAccessException e) {
135 			Prolog.getProvider().getLogger().error(ReflectionUtil.class, e.getMessage(), e);
136 		} catch (NoSuchMethodException e) {
137 			Prolog.getProvider().getLogger().error(ReflectionUtil.class, e.getMessage(), e);
138 		} catch (SecurityException e) {
139 			Prolog.getProvider().getLogger().error(ReflectionUtil.class, e.getMessage(), e);
140 		} catch (IllegalArgumentException e) {
141 			Prolog.getProvider().getLogger().error(ReflectionUtil.class, e.getMessage(), e);
142 		} catch (InvocationTargetException e) {
143 			Prolog.getProvider().getLogger().error(ReflectionUtil.class, e.getMessage(), e);
144 		}
145 		return value;
146 	}

170 	/**
171 	 * Write in a field value given some owner object and object value. This method
172 	 * treated an special case when try write an object array. In this case the
173 	 * array will be a copy with the array component type expected and defined in
174 	 * the filed.
175 	 * 
176 	 * @param field  class field for write object value
177 	 * @param object object field owner where will be wrote the value
178 	 * @param value  value to be write.
179 	 * @since 1.0
180 	 */
181 	public static void writeValue(Field field, Object object, Object value) {
182 		try {
183 
184 			Class<?> clazz = field.getType();
185 			if (value == null) {
186 				if (clazz == boolean.class) {
187 					value = false;
188 				} else if (clazz == int.class) {
189 					value = 0;
190 				} else if (clazz == long.class) {
191 					value = 0L;
192 				} else if (clazz == float.class) {
193 					value = 0F;
194 				} else if (clazz == double.class) {
195 					value = 0D;
196 				}
197 			} else if (value instanceof Object[] && clazz != Object[].class) {
198 				value = castComponentType((Object[]) value, clazz);
199 			}
200 
201 			doAccessibleIfNeed(field);
202 			field.set(object, value);
203 			undoAccessibleIfNeed(field);
204 		} catch (IllegalArgumentException e) {
205 			Prolog.getProvider().getLogger().error(ReflectionUtil.class, e.getMessage(), e);
206 		} catch (IllegalAccessException e) {
207 			Prolog.getProvider().getLogger().error(ReflectionUtil.class, e.getMessage(), e);
208 		}
209 	}

214 	 * 
215 	 * @param array object array to be copied
216 	 * @param type  new component type
217 	 * @return new array of the class type
218 	 * @since 1.0
219 	 */
220 	public static Object castComponentType(Object[] array, Class<?> type) {
221 		Class<? extends Object[]> clazz = (Class<? extends Object[]>) type;
222 		return Arrays.copyOf(array, array.length, clazz);
223 	}

237 	public static Method getDeclaredMethod(Class<? extends Object> class1, String methodName,
238 			Class<?>... parameterTypes) {
239 		Method method = null;
240 		try {
241 			Method[] methods = class1.getDeclaredMethods();
242 			for (Method method2 : methods) {
243 				if (method2.getName().equals(methodName)) {
244 					Class<?>[] types = method2.getParameterTypes();
245 					if (Arrays.equals(types, parameterTypes)) {
246 						method = class1.getDeclaredMethod(methodName, parameterTypes);
247 					} else {
248 						method = method2;
249 					}
250 				}
251 			}
252 		} catch (NoSuchMethodException e) {
253 			Prolog.getProvider().getLogger().error(ReflectionUtil.class, e.getMessage(), e);
254 		} catch (SecurityException e) {
255 			Prolog.getProvider().getLogger().error(ReflectionUtil.class, e.getMessage(), e);
256 		}
257 		return method;
258 	}

260 	/**
261 	 * Invoke a method from some given owner object with given arguments.
262 	 * 
263 	 * @param object    method owner.
264 	 * @param method    method to be invoke.
265 	 * @param arguments signature method arguments required.
266 	 * @return object result of invoke the given method.
267 	 */
268 	public static Object invoke(Object object, Method method, Object... arguments) {
269 		Object ret = null;
270 		try {
271 			doAccessibleIfNeed(method);
272 			ret = method.invoke(object, arguments);
273 			undoAccessibleIfNeed(method);
274 		} catch (IllegalArgumentException e) {
275 			Prolog.getProvider().getLogger().error(ReflectionUtil.class, e.getMessage(), e);
276 		} catch (IllegalAccessException e) {
277 			Prolog.getProvider().getLogger().error(ReflectionUtil.class, e.getMessage(), e);
278 		} catch (InvocationTargetException e) {
279 			Prolog.getProvider().getLogger().error(ReflectionUtil.class, e.getMessage(), e);
280 		}
281 		return ret;
282 	}
