225 	public final boolean isCompliant() {
226 		PrologEngine engine = newEngine();
227 		Set<PrologIndicator> implemented = engine.getBuiltIns();
228 		for (PrologIndicator prologIndicator : ISO_IEC_BUILT_INS) {
229 			if (implemented.contains(prologIndicator)) {
230 				return true;
231 			}
232 		}
233 		return false;
234 	}
