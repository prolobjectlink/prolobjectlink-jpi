224 	protected final String removeQuoted(String functor) {
225 		if (functor != null && functor.startsWith("\'") && functor.endsWith("\'")) {
226 			return functor.substring(1, functor.length() - 1);
227 		}
228 		return functor;
229 	}

231 	public final boolean isCompliant() {
232 		PrologEngine engine = newEngine();
233 		Set<PrologIndicator> implemented = engine.getBuiltIns();
234 		for (PrologIndicator prologIndicator : ISO_IEC_BUILT_INS) {
235 			if (implemented.contains(prologIndicator)) {
236 				return true;
237 			}
238 		}
239 		return false;
240 	}
