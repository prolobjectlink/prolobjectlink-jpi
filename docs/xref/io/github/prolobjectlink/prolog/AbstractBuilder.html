33  package io.github.prolobjectlink.prolog;

35  /**
36   * Partial implementation of Prolog Builder interface. Is an String Builder
37   * wrapper to append the term in string form that compound the final clause
38   * (Fact, Rule or Query).
39   * 
40   * @author Jose Zalacain
41   * @since 1.0
42   */
43  abstract class AbstractBuilder implements PrologBuilder {
44  
45  	protected StringBuilder builder;
46  	protected final PrologEngine engine;
47  	private final PrologProvider provider;
48  
49  	AbstractBuilder(PrologEngine engine) {
50  		this.provider = engine.getProvider();
51  		this.builder = new StringBuilder();
52  		this.engine = engine;
53  	}
54  
55  	protected final void append(Object object) {
56  		builder.append(object);
57  	}
58  
59  	protected final void append(String functor, PrologTerm... arguments) {
60  		if (arguments != null && arguments.length > 0) {
61  			builder.append(provider.newStructure(functor, arguments));
62  		} else {
63  			builder.append(provider.newAtom(functor));
64  		}
65  	}
66  
67  	protected final void append(Object left, String operator, Object right) {
68  		builder.append(left);
69  		append(' ');
70  		builder.append(operator);
71  		append(' ');
72  		builder.append(right);
73  	}
74  
75  	public final PrologEngine getEngine() {
76  		return engine;
77  	}
78  
79  	@Override
80  	public final boolean equals(Object object) {
81  		if (this == object)
82  			return true;
83  		if (object == null)
84  			return false;
85  		if (getClass() != object.getClass())
86  			return false;
87  		DefaultQueryBuilder other = (DefaultQueryBuilder) object;
88  		if (engine == null) {
89  			if (other.engine != null)
90  				return false;
91  		} else if (!engine.equals(other.engine)) {
92  			return false;
93  		}
94  		if (builder == null) {
95  			if (other.builder != null)
96  				return false;
97  		} else if (!builder.toString().equals(other.builder.toString())) {
98  			return false;
99  		}
100 		return true;
101 	}
102 
103 	@Override
104 	public final int hashCode() {
105 		final int prime = 31;
106 		int result = 1;
107 		result = prime * result + ((engine == null) ? 0 : engine.hashCode());
108 		result = prime * result + ((builder == null) ? 0 : builder.hashCode());
109 		return result;
110 	}
111 
112 	@Override
113 	public final String toString() {
114 		return "" + builder + "";
115 	}
