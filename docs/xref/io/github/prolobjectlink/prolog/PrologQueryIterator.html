33  package io.github.prolobjectlink.prolog;
34  
35  import java.util.Collection;
36  import java.util.Iterator;
37  import java.util.NoSuchElementException;
38  
39  /**
40   * Iterator class implementation for Prolog Query
41   * 
42   * @author Jose Zalacain
43   * @since 1.0
44   */
45  final class PrologQueryIterator extends AbstractIterator<Collection<PrologTerm>>
46  		implements Iterator<Collection<PrologTerm>> {
47  
48  	private final PrologQuery query;
49  
50  	/**
51  	 * Create a new query iterator over given PrologQuery.
52  	 * 
53  	 * @param query PrologQuery to iterate
54  	 */
55  	PrologQueryIterator(PrologQuery query) {
56  		this.query = query;
57  	}
58  
59  	/**
60  	 * True if the current Prolog query has more solution, false if not has more
61  	 * solution
62  	 */
63  	public boolean hasNext() {
64  		return query.hasMoreSolutions();
65  	}
66  
67  	/**
68  	 * Next Prolog solution terms
69  	 */
70  	public Collection<PrologTerm> next() {
71  		if (!query.hasMoreSolutions()) {
72  			throw new NoSuchElementException();
73  		}
74  		return query.nextVariablesSolution().values();
75  	}
76  
77  	/**
78  	 * Skip the next solution invoking next solution method
79  	 */
80  	@Override
81  	public void remove() {
82  		query.nextSolution();
83  	}
84  
85  }
