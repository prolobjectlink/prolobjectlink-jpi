52  public abstract class AbstractConsole implements PrologConsole {

54  	private static final String PROLOBJECTLINK = "Prolobjectlink";
55  	private static final String COPYRIHT = " (C)";
56  
57  	private static final InputStream STDIN = System.in;
58  //	private static final OutputStream STDOUT = System.out
59  
60  	// default input stream
61  	private final InputStreamReader input = new InputStreamReader(STDIN);
62  
63  	// buffered reader for read from standard input stream
64  	private final BufferedReader reader = new BufferedReader(input);
65  
66  	// standard output stream
67  //	private final PrintWriter output = System.console().writer()
68  	private final PrintWriter output = new PrintWriter(System.out, true);

73  	public AbstractConsole(PrologProvider provider) {
74  		this.engine = provider.newEngine();
75  	}

77  	public final Map<String, String> getArguments(String[] args) {
78  		final Map<String, String> map = new HashMap<String, String>();
79  		if (args.length > 0) {
80  			Iterator<String> i = new ArrayIterator<String>(args);
81  			String name = i.next();
82  			if (i.hasNext()) {
83  				String value = i.next();
84  				map.put(name, value);
85  			} else {
86  				map.put(name, "");
87  			}
88  		}
89  		return map;
90  	}

93  		output.println("Usage: pllink option [file]");

95  		output.println("	-r	consult/run a prolog file");
96  		output.println("	-v	print the prolog engine version");
97  		output.println("	-n	print the prolog engine name");
98  		output.println("	-l	print the prolog engine license");
99  		output.println("	-i	print the prolog engine information");
100 		output.println("	-a	print the prolog engine about");
101 		output.println("	-e	print the prolog engine enviroment paths");
102 		output.println("	-x	start the prolog engine execution");
103 		output.println("	-w	print the current work directory ");
104 		output.println("	-f	consult a prolog file and save formatted code");
105 		output.println("	-t	test and report integration conditions");
106 		output.println("	-p	print in a file a snapshot of currents predicates");
107 		output.println("	-s	generate .project file for Prolog Development Tool");
108 	}

110 	public final void run(String[] args) {
111 
112 		Map<String, String> m = getArguments(args);
113 		if (!m.isEmpty()) {
114 			if (m.containsKey("-v")) {
115 				output.println(engine.getVersion());
116 			} else if (m.containsKey("-n")) {
117 				output.println(engine.getName());
118 			} else if (m.containsKey("-l")) {
119 				output.println(engine.getLicense());
120 			} else if (m.containsKey("-i")) {
121 				output.print(PROLOBJECTLINK);
122 				output.print(COPYRIHT);
123 				output.print(" ");
124 				output.print(engine.getName());
125 				output.print(" v");
126 				output.println(engine.getVersion());
127 				output.println(engine.getLicense());
128 				output.println(System.getProperty("java.vm.name"));
129 				output.println(System.getProperty("java.vendor"));
130 				output.println(System.getProperty("java.version"));
131 				output.println();
132 			} else if (m.containsKey("-w")) {
133 				try {
134 					output.println("Working directory");
135 					ProtectionDomain p = getClass().getProtectionDomain();
136 					URI d = p.getCodeSource().getLocation().toURI();
137 					output.println(d);
138 				} catch (URISyntaxException e) {
139 					Logger.getLogger(getClass().getName()).log(Level.SEVERE, null, e);
140 				}
141 			} else if (m.containsKey("-e")) {
142 				output.println("Enviroment");
143 				output.println("Class path");
144 				output.println(System.getenv("java.class.path"));
145 				output.println("System path");
146 				output.println(System.getenv("Path"));
147 			} else if (m.containsKey("-a")) {
148 				output.print(PROLOBJECTLINK);
149 				output.print(COPYRIHT);
150 			} else if (m.containsKey("-r")) {
151 				String file = m.get("-r");
152 				output.print("Consult ");
153 				output.println(file);
154 				engine.consult(file);
155 			} else if (m.containsKey("-x")) {
156 				// do nothing silently execution
157 			} else if (m.containsKey("-f")) {
158 				String file = m.get("-r");
159 				output.print("Format ");
160 				output.println(file);
161 				engine.consult(file);
162 				engine.persist(file);
163 			} else if (m.containsKey("-t")) {
164 				List<String> status = engine.verify();
165 				for (String string : status) {
166 					output.println(string);
167 				}
168 			} else if (m.containsKey("-p")) {
169 				String file = m.get("-p");
170 				try {
171 					PrintWriter writter = new PrintWriter(file);
172 					Set<PrologIndicator> set = engine.currentPredicates();
173 					for (PrologIndicator prologIndicator : set) {
174 						writter.println(prologIndicator);
175 					}
176 					writter.close();
177 				} catch (FileNotFoundException e) {
178 					Logger.getLogger(getClass().getName()).log(Level.SEVERE, null, e);
179 					System.exit(1);
180 				}
181 			} else {
182 				printUsage();
183 				System.exit(1);
184 			}
185 
186 			try {
187 
188 				String queryString;
189 				output.print("?- ");
190 				output.flush();
191 				queryString = reader.readLine();
192 
193 				while (true) {
194 
195 					if (!queryString.equals("")) {
196 						output.println();
197 
198 						if (queryString.lastIndexOf('.') == queryString.length() - 1) {
199 							queryString = queryString.substring(0, queryString.length() - 1);
200 						}
201 
202 						PrologQuery query = engine.query(queryString);
203 						if (query.hasSolution()) {
204 							Map<String, PrologTerm> s = query.oneVariablesSolution();
205 							for (Entry<String, PrologTerm> e : s.entrySet()) {
206 								output.println(e.getKey() + " = " + e.getValue());
207 							}
208 							output.println();
209 							output.println("Yes.");
210 						}
211 
212 						else {
213 							output.println("No.");
214 						}
215 
216 						output.println();
217 						output.println();
218 
219 					} else {
220 						output.println("Emty query");
221 						output.println();
222 					}
223 
224 					output.print("?- ");
225 					output.flush();
226 					queryString = reader.readLine();
227 
228 				}
229 
230 			} catch (UnsatisfiedLinkError e) {
231 				output.println("Prolog engine link conditions:");
232 				List<String> status = engine.verify();
233 				for (String string : status) {
234 					output.println(string);
235 				}
236 			} catch (IOException e) {
237 				Logger.getLogger(getClass().getName()).log(Level.SEVERE, null, e);
238 				System.exit(1);
239 			}
240 
241 		} else {
242 			printUsage();
243 			System.exit(1);
244 		}
245 
246 	}
