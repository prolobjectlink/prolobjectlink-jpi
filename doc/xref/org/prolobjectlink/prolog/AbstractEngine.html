101 	public final Map<String, List<PrologClause>> getProgramMap() {
102 		Map<String, List<PrologClause>> m = new HashMap<String, List<PrologClause>>();
103 		for (PrologClause clause : this) {
104 			String key = clause.getIndicator();
105 			List<PrologClause> l = m.get(key);
106 			if (l == null) {
107 				l = new ArrayList<PrologClause>();
108 				l.add(clause);
109 				m.put(key, l);
110 			} else {
111 				l.add(clause);
112 			}
113 		}
114 		return m;
115 	}

226 	protected class PrologProgramIterator extends AbstractIterator<PrologClause> implements Iterator<PrologClause> {
227 
228 		private PrologClause last;
229 		private final Iterator<PrologClause> i;
230 
231 		public PrologProgramIterator(Collection<PrologClause> cls) {
232 			i = cls.iterator();
233 		}
234 
235 		public boolean hasNext() {
236 			return i.hasNext();
237 		}
238 
239 		public PrologClause next() {
240 			if (!i.hasNext()) {
241 				throw new NoSuchElementException();
242 			}
243 			last = i.next();
244 			return last;
245 		}
246 
247 		@Override
248 		public void remove() {
249 			PrologTerm h = last.getHead();
250 			PrologTerm b = last.getBody();
251 			retract(h, b);
252 			i.remove();
253 		}
